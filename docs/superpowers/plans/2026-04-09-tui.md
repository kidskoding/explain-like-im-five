# ELI5 TUI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Go TUI using bubbletea that provides a progressive-reveal learning interface for the eli5 skill, calling the Claude API directly with `.claude/skills/eli5.md` as the system prompt.

**Architecture:** A screen-based router holds the active bubbletea Model and delegates Update/View. Screens communicate back to the router via `NavigateMsg` commands. The Claude client streams tokens into a channel, exposing a `NextToken()` tea.Cmd that the active screen recursively calls in its Update loop. The vault package reads/writes `library/` using a fixed default of `~/.eli5/library`, overridable via `ELI5_VAULT`.

**Tech Stack:** Go 1.22+, `github.com/charmbracelet/bubbletea`, `github.com/charmbracelet/lipgloss`, `github.com/anthropics/anthropic-sdk-go`

package vault

import (
	"fmt"
	"os"
	"path/filepath"
	"sort"
	"strings"
	"time"
)

// FrontMatter holds the YAML front matter for a vault entry.
type FrontMatter struct {
	Topic  string   // for topics
	Course string   // for courses
	Date   string
	Tags   []string
}

// Entry represents a single vault file.
type Entry struct {
	Slug     string
	Topic    string
	Course   string
	Tags     []string
	ModTime  time.Time
	IsCourse bool
}

// Vault manages reading and writing library files.
type Vault struct {
	dir string
}

// New creates a Vault rooted at dir.
func New(dir string) *Vault {
	return &Vault{dir: dir}
}

// DefaultDir returns the vault directory: $ELI5_VAULT or ~/.eli5/library.
func DefaultDir() string {
	if v := os.Getenv("ELI5_VAULT"); v != "" {
		return v
	}
	home, err := os.UserHomeDir()
	if err != nil {
		return "./library"
	}
	return filepath.Join(home, ".eli5", "library")
}

// Save writes a topic explanation to library/<slug>.md.
func (v *Vault) Save(slug, content string, fm FrontMatter) error {
	if err := os.MkdirAll(v.dir, 0755); err != nil {
		return fmt.Errorf("creating vault dir: %w", err)
	}
	body := renderFile(fm, content)
	return os.WriteFile(filepath.Join(v.dir, slug+".md"), []byte(body), 0644)
}

// SaveCourse writes a course guide to library/courses/<slug>.md.
func (v *Vault) SaveCourse(slug, content string, fm FrontMatter) error {
	coursesDir := filepath.Join(v.dir, "courses")
	if err := os.MkdirAll(coursesDir, 0755); err != nil {
		return fmt.Errorf("creating courses dir: %w", err)
	}
	body := renderFile(fm, content)
	return os.WriteFile(filepath.Join(coursesDir, slug+".md"), []byte(body), 0644)
}

// Read returns the raw file content for a slug.
// For topics: slug is the filename stem. For courses: prefix with "courses/".
func (v *Vault) Read(slug string) (string, error) {
	path := filepath.Join(v.dir, slug+".md")
	b, err := os.ReadFile(path)
	if err != nil {
		// try courses/
		path = filepath.Join(v.dir, "courses", slug+".md")
		b, err = os.ReadFile(path)
		if err != nil {
			return "", err
		}
	}
	return string(b), nil
}

// List returns all vault entries sorted by modification time (newest first), max 50.
func (v *Vault) List() ([]Entry, error) {
	var entries []Entry

	// topics
	topicFiles, _ := filepath.Glob(filepath.Join(v.dir, "*.md"))
	for _, f := range topicFiles {
		e, err := parseEntry(f, false)
		if err == nil {
			entries = append(entries, e)
		}
	}

	// courses
	courseFiles, _ := filepath.Glob(filepath.Join(v.dir, "courses", "*.md"))
	for _, f := range courseFiles {
		e, err := parseEntry(f, true)
		if err == nil {
			entries = append(entries, e)
		}
	}

	sort.Slice(entries, func(i, j int) bool {
		return entries[i].ModTime.After(entries[j].ModTime)
	})

	if len(entries) > 50 {
		entries = entries[:50]
	}
	return entries, nil
}

// Search filters entries by query (matches slug and tags, case-insensitive).
func (v *Vault) Search(query string, entries []Entry) []Entry {
	q := strings.ToLower(query)
	var results []Entry
	for _, e := range entries {
		if strings.Contains(strings.ToLower(e.Slug), q) {
			results = append(results, e)
			continue
		}
		for _, tag := range e.Tags {
			if strings.Contains(strings.ToLower(tag), q) {
				results = append(results, e)
				break
			}
		}
	}
	return results
}

// renderFile produces the full markdown file content with front matter.
func renderFile(fm FrontMatter, content string) string {
	var sb strings.Builder
	sb.WriteString("---\n")
	if fm.Topic != "" {
		sb.WriteString("topic: " + fm.Topic + "\n")
	}
	if fm.Course != "" {
		sb.WriteString("course: " + fm.Course + "\n")
	}
	sb.WriteString("date: " + fm.Date + "\n")
	if len(fm.Tags) > 0 {
		sb.WriteString("tags: [" + strings.Join(fm.Tags, ", ") + "]\n")
	}
	sb.WriteString("---\n\n")
	sb.WriteString(content)
	return sb.String()
}

// parseEntry reads front matter from a vault file.
func parseEntry(path string, isCourse bool) (Entry, error) {
	info, err := os.Stat(path)
	if err != nil {
		return Entry{}, err
	}
	b, err := os.ReadFile(path)
	if err != nil {
		return Entry{}, err
	}

	slug := strings.TrimSuffix(filepath.Base(path), ".md")
	e := Entry{
		Slug:     slug,
		ModTime:  info.ModTime(),
		IsCourse: isCourse,
	}

	// Parse front matter between --- delimiters
	content := string(b)
	if !strings.HasPrefix(content, "---") {
		return e, nil
	}
	end := strings.Index(content[3:], "---")
	if end < 0 {
		return e, nil
	}
	fm := content[3 : end+3]
	for _, line := range strings.Split(fm, "\n") {
		line = strings.TrimSpace(line)
		if strings.HasPrefix(line, "topic:") {
			e.Topic = strings.TrimSpace(strings.TrimPrefix(line, "topic:"))
		}
		if strings.HasPrefix(line, "course:") {
			e.Course = strings.TrimSpace(strings.TrimPrefix(line, "course:"))
		}
		if strings.HasPrefix(line, "tags:") {
			raw := strings.TrimSpace(strings.TrimPrefix(line, "tags:"))
			raw = strings.Trim(raw, "[]")
			for _, t := range strings.Split(raw, ",") {
				if tag := strings.TrimSpace(t); tag != "" {
					e.Tags = append(e.Tags, tag)
				}
			}
		}
	}
	return e, nil
}

package claude

import (
	"context"
	"errors"
	"fmt"
	"os"
	"strings"

	anthropic "github.com/anthropics/anthropic-sdk-go"
	"github.com/anthropics/anthropic-sdk-go/option"
	tea "github.com/charmbracelet/bubbletea"
)

const defaultModel = "claude-sonnet-4-6"

// Section headers the TUI detects in topic mode.
var sectionHeaders = []string{
	"### 1.",
	"### 2.",
	"### 3.",
	"### 4.",
}

// Msg types emitted as bubbletea messages during streaming.
type TokenMsg string
type SectionCompleteMsg struct{ Num int }
type StreamDoneMsg struct{}
type StreamErrMsg struct{ Err error }

// Client wraps the Anthropic SDK for streaming ELI5 responses.
type Client struct {
	api          *anthropic.Client
	systemPrompt string
	model        string
}

// New creates a Client. systemPromptPath is the path to the skill file;
// if empty or unreadable, falls back to the hardcoded prompt.
func New(systemPromptPath string) (*Client, error) {
	key := os.Getenv("ANTHROPIC_API_KEY")
	if key == "" {
		return nil, errors.New("ANTHROPIC_API_KEY is not set")
	}

	model := os.Getenv("ELI5_MODEL")
	if model == "" {
		model = defaultModel
	}

	prompt := loadSystemPrompt(systemPromptPath)

	api := anthropic.NewClient(option.WithAPIKey(key))
	return &Client{api: api, systemPrompt: prompt, model: model}, nil
}

// Stream starts a streaming API call and returns a *Stream for reading tokens.
// Call stream.NextToken() in your bubbletea Update loop to receive TokenMsg values.
func (c *Client) Stream(prompt string) *Stream {
	ch := make(chan tea.Msg, 256)
	s := &Stream{ch: ch}

	go func() {
		defer close(ch)
		ctx := context.Background()

		stream := c.api.Messages.NewStreaming(ctx, anthropic.MessageNewParams{
			Model:     anthropic.F(c.model),
			MaxTokens: anthropic.F(int64(8192)),
			System: anthropic.F([]anthropic.TextBlockParam{
				{Type: anthropic.F(anthropic.TextBlockParamTypeText), Text: anthropic.F(c.systemPrompt)},
			}),
			Messages: anthropic.F([]anthropic.MessageParam{
				anthropic.NewUserMessage(anthropic.NewTextBlock(prompt)),
			}),
		})

		buf := ""
		completedSections := 0

		for stream.Next() {
			event := stream.Current()
			// Extract text delta — verify exact field path with anthropic-sdk-go docs
			if event.Type == anthropic.MessageStreamEventTypeContentBlockDelta {
				if event.Delta.Type == anthropic.ContentBlockDeltaEventDeltaTypeTextDelta {
					token := event.Delta.Text
					buf += token
					ch <- TokenMsg(token)

					// Detect section boundaries
					newCompleted := CountCompletedSections(buf)
					for completedSections < newCompleted {
						completedSections++
						ch <- SectionCompleteMsg{Num: completedSections}
					}
				}
			}
		}

		if err := stream.Err(); err != nil {
			ch <- StreamErrMsg{Err: err}
			return
		}
		ch <- StreamDoneMsg{}
	}()

	return s
}

// Stream holds the channel from a streaming API call.
type Stream struct {
	ch <-chan tea.Msg
}

// NextToken returns a tea.Cmd that reads the next message from the stream.
// Call this in your Update loop after receiving a TokenMsg or SectionCompleteMsg.
func (s *Stream) NextToken() tea.Cmd {
	return func() tea.Msg {
		msg, ok := <-s.ch
		if !ok {
			return StreamDoneMsg{}
		}
		return msg
	}
}

// CountCompletedSections counts how many sections are complete in buf.
// A section is complete when the NEXT section's header has appeared.
func CountCompletedSections(buf string) int {
	count := 0
	for i := 1; i < len(sectionHeaders); i++ {
		if strings.Contains(buf, sectionHeaders[i]) {
			count = i
		}
	}
	return count
}

// ExtractSection returns the content of section num (1-4) from buf.
// Content runs from the section header to the next section header (or end of buf).
func ExtractSection(buf string, num int) string {
	if num < 1 || num > len(sectionHeaders) {
		return ""
	}
	header := sectionHeaders[num-1]
	start := strings.Index(buf, header)
	if start < 0 {
		return ""
	}
	// Skip past the header line
	start = strings.Index(buf[start:], "\n")
	if start < 0 {
		return ""
	}
	content := strings.TrimSpace(buf[start+1:])

	// Trim at next section header
	if num < len(sectionHeaders) {
		nextHeader := sectionHeaders[num]
		if end := strings.Index(content, nextHeader); end >= 0 {
			content = strings.TrimSpace(content[:end])
		}
	}
	return content
}

// loadSystemPrompt reads the skill file. Falls back to embedded prompt if unavailable.
func loadSystemPrompt(path string) string {
	if path != "" {
		b, err := os.ReadFile(path)
		if err == nil {
			// Strip YAML frontmatter (between --- delimiters)
			s := string(b)
			if strings.HasPrefix(s, "---") {
				end := strings.Index(s[3:], "---")
				if end >= 0 {
					return strings.TrimSpace(s[end+6:])
				}
			}
			return s
		}
	}
	return fallbackPrompt
}

// fallbackPrompt is used when .claude/skills/eli5.md cannot be read.
const fallbackPrompt = `You are an expert explainer. Explain topics using this structure:
### 1. The Analogy
### 2. What's Actually Happening
### 3. Go Deeper
### 4. Now You Can Understand

Keep explanations concrete, deep, and never condescending.`

// TopicPrompt formats the user message for single topic mode.
func TopicPrompt(topic string) string {
	return fmt.Sprintf("/eli5 %s", topic)
}

// CoursePrompt formats the user message for course mode (full guide).
func CoursePrompt(cert string) string {
	return fmt.Sprintf("/eli5 course %s", cert)
}

package router

import (
	tea "github.com/charmbracelet/bubbletea"
)

// ScreenID identifies which screen is active.
type ScreenID int

const (
	ScreenHome ScreenID = iota
	ScreenTopicReveal
	ScreenCourseEnroll
	ScreenCourseChapter
	ScreenVaultBrowser
)

// NavigateMsg is sent by screens to request navigation.
// Data carries screen-specific context (e.g., topic string, cert name).
type NavigateMsg struct {
	Target ScreenID
	Data   any
}

// Navigate returns a tea.Cmd that emits a NavigateMsg.
func Navigate(target ScreenID, data any) tea.Cmd {
	return func() tea.Msg {
		return NavigateMsg{Target: target, Data: data}
	}
}

// Screen is implemented by every screen model.
type Screen interface {
	tea.Model
}

// ScreenFactory creates a fresh Screen for a given target and data.
type ScreenFactory func(target ScreenID, data any) Screen

// Router is the root bubbletea model. It holds the active screen
// and delegates Update/View to it.
type Router struct {
	current Screen
	factory ScreenFactory
	width   int
	height  int
}

// New creates a Router with the given initial screen and factory.

	return &Router{current: initial, factory: factory}
}

func (r *Router) Init() tea.Cmd {
	return r.current.Init()
}

func (r *Router) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
	switch msg := msg.(type) {
	case NavigateMsg:
		r.current = r.factory(msg.Target, msg.Data)
		return r, tea.Batch(
			r.current.Init(),
			// pass window size to new screen
			func() tea.Msg { return tea.WindowSizeMsg{Width: r.width, Height: r.height} },
		)
	case tea.WindowSizeMsg:
		r.width = msg.Width
		r.height = msg.Height
	}
	newScreen, cmd := r.current.Update(msg)
	r.current = newScreen.(Screen)
	return r, cmd
}

func (r *Router) View() string {
	return r.current.View()
}

			if m.input == "" {

			}

			if m.cursor >= 0 && m.cursor < len(m.recent) {

			if m.input == "" {

			}

// parseSections extracts section content from the accumulated buffer.
func (m *Model) parseSections() {

		}

// liveSection returns the content of the currently-streaming section.
func liveSection(buf string, sectionNum int) string {
	headers := []string{"### 1.", "### 2.", "### 3.", "### 4."}
	if sectionNum < 1 || sectionNum > 4 {
		return ""
	}
	header := headers[sectionNum-1]
	start := strings.Index(buf, header)
	if start < 0 {
		return buf // no header yet, show everything
	}
	nl := strings.Index(buf[start:], "\n")
	if nl < 0 {
		return ""
	}
	return strings.TrimSpace(buf[start+nl+1:])
}

func (m Model) autoSave() tea.Cmd {
	return func() tea.Msg {
		return savedMsg("Saved to library/" + toSlug(m.topic) + ".md  ✓")
	}
}

// toSlug converts a topic to a kebab-case slug.
func toSlug(topic string) string {
	filler := map[string]bool{"how": true, "why": true, "what": true, "is": true, "the": true, "a": true}
	words := strings.Fields(strings.ToLower(topic))
	var kept []string
	for _, w := range words {
		w = strings.Map(func(r rune) rune {
			if (r >= 'a' && r <= 'z') || (r >= '0' && r <= '9') || r == '-' {
				return r
			}
			return '-'
		}, w)
		if !filler[w] {
			kept = append(kept, w)
		}
	}
	return strings.Join(kept, "-")
}

// parseDomains extracts domain names from the Claude response.
// Looks for numbered list items like "1. Domain Name" or bold "**Domain Name**".
func parseDomains(buf string) []string {
	var domains []string
	for _, line := range strings.Split(buf, "\n") {
		line = strings.TrimSpace(line)
		// Match "1. Domain Name (weight: ...)" pattern
		if len(line) > 2 && line[1] == '.' && line[0] >= '1' && line[0] <= '9' {
			d := strings.TrimSpace(line[2:])
			// Strip weight annotations like "(exam weight: XX%)"
			if idx := strings.Index(d, "("); idx > 0 {
				d = strings.TrimSpace(d[:idx])
			}
			if d != "" {
				domains = append(domains, d)
			}
		}
	}
	return domains
}

func itoa(n int) string {

			return "0"
		}
		s := ""
		for n > 0 {
			s = string(rune('0'+n%10)) + s
			n /= 10
		}
		return s
	}())

		if m.stream != nil {
			return m, m.stream.NextToken()
		}

	case claude.SectionCompleteMsg:
		m.completed = msg.Num
		if m.revealed == 0 {
			m.revealed = 1
			m.state = stateReading
		}
		if m.stream != nil {
			return m, m.stream.NextToken()
		}

	case claude.StreamDoneMsg:
		m.completed = 4

			m.revealed = 1
		}
		m.state = stateReading

	case savedMsg:
		m.saveMsg = string(msg)

	case tea.KeyMsg:
		switch msg.String() {
		case "q", "esc":
			return m, router.Navigate(router.ScreenCourseEnroll, m.data.Cert)
		case " ":
			if m.state == stateReading && m.revealed < m.completed {
				m.revealed++
				if m.revealed == 4 {

				prev := New(m.data, m.domainIndex-1, m.client)

			}

type savedMsg string

func toSlug(s string) string {
	filler := map[string]bool{"how": true, "why": true, "what": true, "is": true, "the": true, "a": true}
	words := strings.Fields(strings.ToLower(s))
	var kept []string
	for _, w := range words {
		w = strings.Map(func(r rune) rune {
			if (r >= 'a' && r <= 'z') || (r >= '0' && r <= '9') {
				return r
			}
			return '-'
		}, w)
		if !filler[w] && w != "" && w != "-" {
			kept = append(kept, w)
		}
	}
	return strings.Join(kept, "-")
}

func (m Model) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
	switch msg := msg.(type) {
	case tea.WindowSizeMsg:
		m.width = msg.Width
		m.height = msg.Height

	case tea.KeyMsg:
		if m.searching {
			switch msg.String() {
			case "esc", "enter":
				m.searching = false
			case "backspace":
				if len(m.query) > 0 {
					m.query = m.query[:len(m.query)-1]
					m.applyFilter()
				}

				m.query += msg.String()

			}
			return m, nil
		}

		switch msg.String() {
		case "q", "esc":
			return m, router.Navigate(router.ScreenHome, nil)
		case "/":
			m.searching = true
		case "up":
			if m.cursor > 0 {
				m.cursor--
			}
		case "down":
			if m.cursor < len(m.filtered)-1 {
				m.cursor++
			}
		case "enter":
			if len(m.filtered) == 0 {
				return m, nil
			}
			entry := m.filtered[m.cursor]
			if entry.IsCourse {
				return m, router.Navigate(router.ScreenCourseEnroll, entry.Slug)
			}
			return m, router.Navigate(router.ScreenTopicReveal, entry.Slug)
		}
	}
	return m, nil
}

func (m Model) View() string {
	var sb strings.Builder
	sb.WriteString(titleStyle.Render("Vault") + "\n\n")

	if m.searching {
		sb.WriteString(activeStyle.Render("/"+m.query+"█") + "\n\n")

		sb.WriteString(subtleStyle.Render("/"+m.query) + "\n\n")

			sb.WriteString(subtleStyle.Render("No results for \"" + m.query + "\"") + "\n")
		} else {
			sb.WriteString(subtleStyle.Render("No explanations yet — go learn something.") + "\n")
		}
	} else {
		for i, e := range m.filtered {
			name := e.Topic
			if name == "" {
				name = e.Course
			}
			if name == "" {
				name = e.Slug
			}
			prefix := "  "
			if i == m.cursor {
				prefix = "› "
				sb.WriteString(activeStyle.Render(prefix+name) + "\n")
			} else {
				sb.WriteString(subtleStyle.Render(prefix+name) + "\n")
			}
		}
	}

	sb.WriteString("\n" + hintStyle.Render("[enter] open  [/] search  [q] back"))

}

func (m *Model) applyFilter() {
	if m.query == "" {
		m.filtered = m.all
		return
	}
	q := strings.ToLower(m.query)
	var results []vault.Entry
	for _, e := range m.all {
		if strings.Contains(strings.ToLower(e.Slug), q) ||
			strings.Contains(strings.ToLower(e.Topic), q) ||
			strings.Contains(strings.ToLower(e.Course), q) {
			results = append(results, e)
			continue
		}
		for _, tag := range e.Tags {
			if strings.Contains(strings.ToLower(tag), q) {
				results = append(results, e)
				break
			}
		}
	}
	m.filtered = results
	m.cursor = 0
}

func main() {
	// Find skill file — check project dir first, then ~/.claude/skills/
	skillPath := findSkillFile()

	// Init Claude client
	claudeClient, err := claude.New(skillPath)
	apiErr := ""
	if err != nil {
		apiErr = err.Error()
		claudeClient = nil
	}

	// Init vault
	vaultDir := vault.DefaultDir()
	v := vault.New(vaultDir)

	// Load recent entries for home screen
	recent, _ := v.List()
	if len(recent) > 10 {
		recent = recent[:10]
	}

	// Build screen factory
	factory := func(target router.ScreenID, data any) router.Screen {
		switch target {
		case router.ScreenHome:
			entries, _ := v.List()
			if len(entries) > 10 {
				entries = entries[:10]
			}
			return home.New(entries, apiErr)

		case router.ScreenTopicReveal:
			topic, _ := data.(string)
			if topic == "" {
				return home.New(recent, apiErr)
			}
			// Check if it's a vault slug (readonly) or a new topic
			if content, err := v.Read(topic); err == nil {
				return topic_reveal.NewReadonly(topic, content)
			}
			return topic_reveal.New(topic, claudeClient)

		case router.ScreenCourseEnroll:
			cert, _ := data.(string)
			return course_enroll.New(cert, claudeClient)

		case router.ScreenCourseChapter:
			switch d := data.(type) {
			case course_chapter.Model:
				return d
			case course_enroll.CourseData:
				return course_chapter.New(d, 0, claudeClient)
			}
			return home.New(recent, apiErr)

		case router.ScreenVaultBrowser:
			entries, _ := v.List()
			return vault_browser.New(entries)

		default:
			return home.New(recent, apiErr)
		}
	}

	initial := home.New(recent, apiErr)
	r := router.New(initial, factory)

	p := tea.NewProgram(r, tea.WithAltScreen())
	if _, err := p.Run(); err != nil {
		fmt.Fprintf(os.Stderr, "Error: %v\n", err)
		os.Exit(1)
	}
}

// findSkillFile looks for .claude/skills/eli5.md in the current directory,
// then in ~/.claude/skills/.
func findSkillFile() string {
	candidates := []string{
		".claude/skills/eli5.md",
	}
	if home, err := os.UserHomeDir(); err == nil {
		candidates = append(candidates, filepath.Join(home, ".claude", "skills", "eli5.md"))
	}
	for _, p := range candidates {
		if _, err := os.Stat(p); err == nil {
			return p
		}
	}
	return "" // falls back to embedded prompt in claude.New
}

package slugify

import "strings"

// Slug converts a human-readable string to a kebab-case slug,
// dropping standalone filler words.
func Slug(s string) string {
	filler := map[string]bool{"how": true, "why": true, "what": true, "is": true, "the": true, "a": true}
	words := strings.Fields(strings.ToLower(s))
	var kept []string
	for _, w := range words {
		w = strings.Map(func(r rune) rune {
			if (r >= 'a' && r <= 'z') || (r >= '0' && r <= '9') {
				return r
			}
			return '-'
		}, w)
		// trim leading/trailing dashes from word
		w = strings.Trim(w, "-")
		if w != "" && !filler[w] {
			kept = append(kept, w)
		}
	}
	return strings.Join(kept, "-")
}
